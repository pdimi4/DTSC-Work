# Fit regression model with interaction
# Note: statsmodels automatically treats string variables as categorical

# Plot: Interaction lines
plt.figure(figsize=(9, 6))
x = np.linspace(df["StudyHours"].min(), df["StudyHours"].max(), 100)

for group in ["Yes", "No"]:
    pred = model.predict(pd.DataFrame({
        "StudyHours": x,
        "Tutoring": group
    }))
    plt.plot(x, pred, label=f"Tutoring = {group}")

# Overlay raw data
